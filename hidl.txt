# 출력 디렉터리
OUTDIR=hidl_test/1.0/gen
mkdir -p "$OUTDIR"

# 헤더 생성
hidl-gen \
  -Lc++-headers \
  -o "$OUTDIR" \
  -r vendor.example.led:hidl_test \
  -r android.hidl:system/libhidl/transport \
  vendor.example.led@1.0

# 소스 생성
hidl-gen \
  -Lc++-sources \
  -o "$OUTDIR" \
  -r vendor.example.led:hidl_test \
  -r android.hidl:system/libhidl/transport \
  vendor.example.led@1.0
